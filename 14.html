List the names of scholarships for OBC category. MATCH (s:Scholarship) WHERE s.category_eligible = 'OBC' RETURN s.name AS OBC_Scholarships; b. Count number of students benefitted by ____ scholarship in year 2020-2021 MATCH (st:Student)-[b:BENEFITTED_FROM]->(sc:Scholarship) WHERE sc.name = 'OBC Merit Scholarship' AND b.year = '2020-2021' RETURN COUNT(st) AS Benefitted_Students; c. Update the income limit for a given scholarship MATCH (s:Scholarship {name:'OBC Merit Scholarship'}) SET s.income_limit = 400000 RETURN s; d. List the most popular scholarship MATCH (st:Student)-[:BENEFITTED_FROM]->(sc:Scholarship) WITH sc, COUNT(st) AS cnt RETURN sc.name AS Scholarship, cnt ORDER BY cnt DESC LIMIT 1;
